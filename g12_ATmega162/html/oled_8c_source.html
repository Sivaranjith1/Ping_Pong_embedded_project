129 void oled_init(void){
130  xmem_write(0xAE, OLED_BASE_ADDRESS_COMMAND); // display off
131  xmem_write(0xA1, OLED_BASE_ADDRESS_COMMAND); //segment remap
132  xmem_write(0xDA, OLED_BASE_ADDRESS_COMMAND); //common pads hardware: alternative
133  xmem_write(0x12, OLED_BASE_ADDRESS_COMMAND);
134  xmem_write(0xC8, OLED_BASE_ADDRESS_COMMAND); //common output scan direction:com63~com0
135  xmem_write(0xA8, OLED_BASE_ADDRESS_COMMAND); //multiple, x ration mode:63
136  xmem_write(0x3F, OLED_BASE_ADDRESS_COMMAND);
137  xmem_write(0xD5, OLED_BASE_ADDRESS_COMMAND); //display divide ratio/osc. freq. mode
138  xmem_write(0x80, OLED_BASE_ADDRESS_COMMAND);
139  xmem_write(OLED_BRIGHTNESS, OLED_BASE_ADDRESS_COMMAND); //contrast control
140  xmem_write(0x50, OLED_BASE_ADDRESS_COMMAND);
141  xmem_write(0xD9, OLED_BASE_ADDRESS_COMMAND); //set pre-charge period
142  xmem_write(0x21, OLED_BASE_ADDRESS_COMMAND);
143  xmem_write(0x20, OLED_BASE_ADDRESS_COMMAND); //Set Memory Addressing Mode
144  xmem_write(0x02, OLED_BASE_ADDRESS_COMMAND);
145  xmem_write(0xDB, OLED_BASE_ADDRESS_COMMAND); //VCOM deselect level mode
146  xmem_write(0x30, OLED_BASE_ADDRESS_COMMAND);
147  xmem_write(0xAD, OLED_BASE_ADDRESS_COMMAND); //master configuration
148  xmem_write(0x00, OLED_BASE_ADDRESS_COMMAND);
149  xmem_write(0xA4, OLED_BASE_ADDRESS_COMMAND); //out follows RAM content
150  xmem_write(0xA6, OLED_BASE_ADDRESS_COMMAND); //set normal display
151  xmem_write(0xAF, OLED_BASE_ADDRESS_COMMAND); // display on
152  oled_reset();
153 }
