59  // Write phase: Immediately check that the correct value was stored
60  srand(seed);

62  uint8_t some_value = rand();

64  uint8_t retreived_value = ext_ram[i];

68  }
69  }
70  // Retrieval phase: Check that no values were changed during or after the write phase
71  srand(seed);
72  // reset the PRNG to the state it had before the write phase
